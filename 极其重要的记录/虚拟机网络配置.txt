网卡配置修改注意项：
vim /etc/resolv.conf
#修改 
nameserver=192.168.47.2   # 网关IP

vim /etc/sysconfig/network
NETWORKING=yes
HOSTNAME=localhost.localdomain

vim /etc/sysconfig/network-scripts/ifcfg-ens33
#添加
IPADDR=192.168.47.*
NETMASK=255.255.255.0
GATEWAY=192.168.47.2
DNS1=192.168.47.2
#修改
BOOTPROTO="static"
ONBOOT="yes"
#重启网卡
service network restart

#设置hosts
vim /etc/hosts
